kind: Cluster
apiVersion: kind.x-k8s.io/v1alpha4
containerdConfigPatches:
  - |-
    [plugins."io.containerd.grpc.v1.cri".containerd.runtimes.runc.options]
      SystemdCgroup = true
nodes:
  - role: control-plane
  - role: worker
    kubeadmConfigPatches:
      - |
        kind: JoinConfiguration
        nodeRegistration:
          name: mycluster-app-1
  - role: worker
    kubeadmConfigPatches:
      - |
        kind: JoinConfiguration
        nodeRegistration:
          name: mycluster-app-2
2. Create the Cluster:Bash# Use a specific image for consistency and stability
kind create cluster --name k8s-local-lab-multi --image kindest/node:v1.33.4 --config config.yaml
kubectl get nodes
